# 自动点击标签

<script>
    //自动点击的方法 clickId 点击事件的id属性值  字符串类型
    autoClick(clickId="favicon")
    function autoClick(clickId) {
        // IE
        if (document.all) {
            document.getElementById(clickId).click();
        }
        // 其它浏览器
        else {
            const e = document.createEvent("MouseEvents");
            e.initEvent("click", true, true);
            document.getElementById(clickId).dispatchEvent(e);
        }
    }
</script>

# JS正则表达式

^ 匹配一个输入或一行的开头，/^a/匹配"an A"，而不匹配"An a"
$ 匹配一个输入或一行的结尾，/a$/匹配"An a"，而不匹配"an A"
* 匹配前面元字符0次或多次，/ba*/将匹配b,ba,baa,baaa
+ 匹配前面元字符1次或多次，/ba*/将匹配ba,baa,baaa
? 匹配前面元字符0次或1次，/ba*/将匹配b,ba
(x) 匹配x保存x在名为$1...$9的变量中
x|y 匹配x或y
{n} 精确匹配n次
{n,} 匹配n次以上
{n,m} 匹配n-m次
[xyz] 字符集(character set)，匹配这个集合中的任一一个字符(或元字符)
[^xyz] 不匹配这个集合中的任何一个字符
[\b] 匹配一个退格符
\b 匹配一个单词的边界
\B 匹配一个单词的非边界
\cX 这儿，X是一个控制符，/\cM/匹配Ctrl-M
\d 匹配一个字数字符，/\d/ = /[0-9]/
\D 匹配一个非字数字符，/\D/ = /[^0-9]/
\n 匹配一个换行符
\r 匹配一个回车符
\s 匹配一个空白字符，包括\n,\r,\f,\t,\v等
\S 匹配一个非空白字符，等于/[^\n\f\r\t\v]/
\t 匹配一个制表符
\v 匹配一个重直制表符
\w 匹配一个可以组成单词的字符(alphanumeric，这是我的意译，含数字)，包括下划线，如[\w]匹配"$5.98"中的5，等于[a-zA-Z0-9]
\W 匹配一个不可以组成单词的字符，如[\W]匹配"$5.98"中的$，等于[^a-zA-Z0-9]。

# 利用array.map()和array.find()或者arr.findIndex

// var clearDuplicate1 = (arr, key) => Array.from(new Set(arr.map(e => e[key]))).map(e => arr.find(x => x[key] == e));
const clearDuplicate1 = (arr, key) =>Array.from(new Set(arr.map(e => e[key]))).map(e => arr.find(x => x[key] == e))
console.log(clearDuplicate1(arr,'url'))

// var clearDuplicate2 = (arr, key) => Array.from(new Set(arr.map(e => e[key]))).map(e => arr.findIndex(x => x[key] == e)).map(e => arr[e])
const clearDuplicate2 = (arr, key) => Array.from(new Set(arr.map(e => e[key]))).map(e => arr.findIndex(x => x[key] == e)).map(e => arr[e])
console.log(clearDuplicate2(arr,'url'))

# 利用arr.reduce 和 array.find 和 array.concat

const clearDuplicate3 = (arr, key) => arr.reduce((acc, current) => {
    return acc.find(item => item[key] === current[key]) ? acc : acc.concat([current]) 
}, []);
console.log(clearDuplicate3(arr,'url'))
利用array.filter和set.has 或者 对象访问属性的方法
const clearDuplicate4 = (arr, key) => {
    const set = new Set()
    return arr.filter(e => {
        const duplicate = set.has(e[key])
        set.add(e[key])
        return !duplicate
    })
}
console.log(clearDuplicate4(arr,'url'))

const clearDuplicate42 = (arr, key) => {
    const obj = new Object()
    return arr.filter(e => {
        return obj[e[key]] ? false:obj[e[key]] = true
    })
}
console.log(clearDuplicate42(arr,'url'))
利用array.filter 和array.map、indexof
const clearDuplicate5 = (arr, key) => arr.filter((ele, index, a) => (a.map(e => e[key]).indexOf(ele[key])) ===index)
console.log(clearDuplicate5(arr,'url'))

# 利用对象访问属性的方法，判断对象中是否存在key

const result = [];
const obj = {};
for (let i = 0; i < arr.length; i++) {
    if (!obj[arr[i].url]) {
        if (arr[i].url){//去空
            result.push(arr[i]);
        };
        obj[arr[i].url] = true;
    }
}
console.log(result);

# 利用reduce方法遍历数组,reduce第一个参数是遍历需要执行的函数，第二个参数是item的初始值

const obj2 = {};
arr = arr.reduce(function (item, next) {
    obj2[next.url] ? '' : obj2[next.url] = true && item.push(next);
    return item;
}, []);
console.log(arr);
